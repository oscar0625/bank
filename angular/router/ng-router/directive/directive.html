    内置指令
        ng-app
            html调用: ng-app='模块名'
            js声明: angular.module('模块名',[])
            如:
                angular.module('app', [])

        ng-controller
            在html中调用: ng-controller="控制器名"
            js声明: 模块.controller('控制器名', [依赖1,依赖2,...function(参数1,参数2){}])
            如:
                angular.module('app', [])
                    .controller('mainCtrl', ['$scope', function (scope) {})

        ng-repeat
            遍历输出html模板
            如:
                <div class="well" ng-repeat="i in list">
                    <p>{{i.text}}</p>
                    <p>转@{{i.author}}:'{{i.content}}'</p>
                </div>
